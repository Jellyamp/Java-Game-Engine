#version 400 core

// Specify the inputs from the program
in vec3 position;

// Specify the outputs for the fragment shader
out vec3 colour;

void main(void) {
	
	// Set the OpenGL position to the input position
	gl_Position = vec4(position, 1.0);
	// Set the fragment shader output color to the interpolation of each pixel based on the distance of each vertex
	colour = vec3(position.x+0.5, 1.0, position.y+0.5);
}
